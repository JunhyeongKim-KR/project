module ref_counter (
  input  logic clk, rst_n,
  output logic out
);
  logic [1:0] cnt;
  always_ff @(posedge clk or negedge rst_n)
    if (!rst_n) cnt <= 2'b00;
    else        cnt <= cnt + 1;

  assign out = (cnt == 2'b11);
endmodule

module impl_counter (
  input  logic clk, rst_n,
  output logic out
);
  logic [1:0] cnt;
  always_ff @(posedge clk or negedge rst_n)
    if (!rst_n) cnt <= 2'b00;
    else        cnt <= cnt + 1;

  assign out = cnt[1] & cnt[0];
endmodule
