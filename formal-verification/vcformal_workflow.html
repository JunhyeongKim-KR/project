Reference: data_proc.sv (직관적 구현)

sel에 따라 덧셈/뺄셈을 선택하는 if/else 구현

module data_proc (
  input  logic       clk, rst_n, valid_in, sel,
  input  logic [7:0] a, b,
  output logic       valid_out,
  output logic [8:0] result
);
  // Input registers
  logic sel_r, valid_r;
  logic [7:0] a_r, b_r;
  always_ff @(posedge clk ...)
    sel_r <= sel; valid_r <= valid_in;
    a_r <= a; b_r <= b;

  // Comb logic: 직관적 if/else
  always_comb
    if (sel_r)
      comb_result = a_r + b_r;  // 덧셈
    else
      comb_result = a_r - b_r;  // 뺄셈

  // Output registers
  always_ff @(posedge clk ...)
    result <= comb_result;
    valid_out <= valid_r;
endmodule

Implementation: data_proc_opt.sv (최적화 구현)

동일 레지스터 구조, 다른 조합 로직: 2의 보수 기반 통합 가산기

// Comb logic: 최적화 (통합 가산기)
logic [8:0] b_modified;
always_comb begin
  // sel=1: b 그대로, sel=0: ~b+1 (2의 보수 = -b)
  b_modified = sel_r ? {1'b0, b_r} : (~{1'b0, b_r} + 9'd1);
  comb_result = {1'b0, a_r} + b_modified;  // 항상 덧셈 하나로 통합
end

Round-Robin Arbiter (arbiter_rr.sv)

4-port round-robin arbiter. Priority pointer가 grant 후 회전하여 공정한 중재를 보장합니다.

설계 구조

module arbiter_rr (
  input  logic       clk,
  input  logic       rst_n,
  input  logic [3:0] req,
  output logic [3:0] gnt
);

  logic [1:0] ptr;  // priority pointer

  // ptr: grant 후 다음 포트로 회전
  always_ff @(posedge clk or negedge rst_n)
    if (!rst_n) ptr <= 2'd0;
    else if (|gnt)
      case (1'b1)
        gnt[0]: ptr <= 2'd1;
        gnt[1]: ptr <= 2'd2;
        gnt[2]: ptr <= 2'd3;
        gnt[3]: ptr <= 2'd0;
      endcase

  // Combinational grant: ptr 기준 우선순위
  always_comb begin
    gnt = 4'b0000;
    case (ptr)
      2'd0: // port 0 highest
        if (req[0]) gnt = 4'b0001;
        else if (req[1]) gnt = 4'b0010;
        // ... (2,3 순서대로)
      // 2'd1: port 1 highest ...
    endcase
  end
endmodule

SVA Assertions (arbiter_props.sv)

클릭하여 각 assertion의 SVA 코드를 확인하세요.

MUTEX assert
동시에 최대 1개만 grant 가능 (one-hot or zero)
MUTEX: assert property (
  @(posedge clk) disable iff (!rst_n)
  $onehot0(gnt)
);

gnt가 0000, 0001, 0010, 0100, 1000 중 하나임을 보장. 0011 같은 동시 grant는 위반.

NO_SPURIOUS assert
요청 없이 grant 불가 (각 포트별)
generate for (i = 0; i < 4; i++)
  NO_SPURIOUS: assert property (
    @(posedge clk) disable iff (!rst_n)
    gnt[i] |-> req[i]
  );
GRANT_RESPONSE assert
요청이 있으면 0~3 사이클 내 grant 보장
GRANT_RESPONSE: assert property (
  @(posedge clk) disable iff (!rst_n)
  (req != 4'b0000) |-> ##[0:3] (gnt != 4'b0000)
);
COVER_ALL cover
각 포트가 grant 받는 시나리오 도달 가능 확인
COVER_GNT_0: cover property (
  @(posedge clk) disable iff (!rst_n) gnt[0]
);
// COVER_GNT_1, COVER_GNT_2, COVER_GNT_3 동일 패턴

Cover가 hit되면 해당 시나리오에 도달 가능함을 증명. Witness 파형도 제공됩니다.

COVER_ALL_REQ cover
4포트 동시 요청 시나리오
COVER_ALL_REQ: cover property (
  @(posedge clk) disable iff (!rst_n)
  (req == 4'b1111)
);
// Bind statement: 자동으로 arbiter_rr에 checker 연결
bind arbiter_rr arbiter_props u_props (.*);

Spec: 2-Stage Pipeline (madd_pipe2.sv)

out = (a * b) + c, Latency = 2 cycles

module madd_pipe2 (
  input  logic        clk, rst_n, valid_in,
  input  logic [7:0]  a, b, c,
  output logic        valid_out,
  output logic [16:0] result
);
  // Stage 1: multiply + register c
  always_ff @(posedge clk or negedge rst_n)
    if (!rst_n) begin
      mul_reg <= '0; c_reg <= '0; valid_s1 <= 0;
    end else begin
      mul_reg <= a * b;   // 곱셈
      c_reg   <= c;       // c 보존
      valid_s1 <= valid_in;
    end

  // Stage 2: add
  always_ff @(posedge clk or negedge rst_n)
    if (!rst_n) begin
      add_reg <= '0; valid_s2 <= 0;
    end else begin
      add_reg <= mul_reg + c_reg; // 덧셈
      valid_s2 <= valid_s1;
    end

  assign result = add_reg;
  assign valid_out = valid_s2;
endmodule

Impl: 3-Stage Pipeline (madd_pipe3.sv) — Retimed

동일 함수 (a * b) + c, 입력 레지스터 추가로 Latency = 3 cycles

module madd_pipe3 (
  // 동일한 포트 인터페이스
  ...
);
  // Stage 1: 입력 레지스터 (retiming으로 추가)
  always_ff @(posedge clk ...)
    a_reg <= a; b_reg <= b; c_reg <= c;
    valid_s1 <= valid_in;

  // Stage 2: multiply
  always_ff @(posedge clk ...)
    mul_reg <= a_reg * b_reg;
    c_reg2  <= c_reg;
    valid_s2 <= valid_s1;

  // Stage 3: add
  always_ff @(posedge clk ...)
    add_reg <= mul_reg + c_reg2;
    valid_s3 <= valid_s2;

  assign result = add_reg;
  assign valid_out = valid_s3;
endmodule
